            <option value="0.10">10%</</option>

        <div class="result" id="result"></div>

    <script>

        function calculateTip() {

            var bill = document.getElementById("bill").value;

            var tipPercentage = document.getElementById("tip").value;

            

            if (bill === "") {

                document.getElementById("result").innerHTML = "Please enter a bill amount.";

                return;

            }

            

            var tip = bill * tipPercentage;

            var total = parseFloat(bill) + parseFloat(tip);

            

            document.getElementById("result").innerHTML = 

                "Tip: $" + tip.toFixed(2) + "<br>" +

                "Total Amount: $" + total.toFixed(2);

        }

    </script>
